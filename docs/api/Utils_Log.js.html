/* eslint no-console: "off" */
let instance = null;

/**
 * class LogUtil
 */
export class LogUtil {
    constructor() {
        if (!instance) {
            instance = this;
        }
        this._level = LogUtil.LogLevel.INFO;
        return instance;
    }
    print(level, from, message) {
        if (level < this._level) {
            return;
        }
        let date = new Date().toTimeString().replace(/.*(\d{2}:\d{2}:\d{2}).*/, '$1');
        let text = '[' + date + '] ' + from + ': ' + message;
        switch (level) {
            case LogUtil.LogLevel.TRACE:
                text = '[TRACE] ' + text;
                console.trace(text);
                break;
            case LogUtil.LogLevel.DEBUG:
                text = '[DEBUG] ' + text;
                console.debug(text);
                break;
            case LogUtil.LogLevel.INFO:
                text = '[INFO] ' + text;
                console.info(text);
                break;
            case LogUtil.LogLevel.WARNING:
                text = '[WARNING] ' + text;
                console.warn(text);
                break;
            case LogUtil.LogLevel.ERROR:
                text = '[ERROR] ' + text;
                console.error(text);
                break;
            case LogUtil.LogLevel.HIGHLIGHTED:
                text = '[HIGHLIGHTED] ' + text;
                console.log(text);
                break;                
        }
    }

    setLogLevel(level) {
        this._level = level;
    }
}

/**
 * Log level
 * @readonly
 * @enum {number}
 * @static
 * @type LogUtil.LogLevel
 */
LogUtil.LogLevel = {
    TRACE: 0,
    DEBUG: 1,
    INFO: 2,
    WARNING: 3,
    ERROR: 4,
    HIGHLIGHTED: 99
};
