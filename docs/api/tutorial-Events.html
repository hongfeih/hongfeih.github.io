Tutorial: Events

Events

1. PlayStart / PlayEnd

player.addEventListener(AdaptivePlayer.EventType.PlayStart, function (e) {
    var loadTime = e.details.loadTime; // load time
});

player.addEventListener(AdaptivePlayer.EventType.PlayEnd, function (e) {
    if (e.details.reason == AdaptivePlayer.ActionReason.ABORT) {
        // user stop
        // ...
    } else if (e.details.reason == AdaptivePlayer.ActionReason.COMPLETE) {
        // playback completed
        // ...
    }
});

6. SeekRangeChange

player.addEventListener(AdaptivePlayer.EventType.SeekRangeChange, function (e) {
    seekRange.start = e.details.start; // start time of seek range
    seekRange.end = e.details.end; // end time of seek range
    seekRange.ranges = e.details.ranges; // ranges in case of mulitple periods/discontinuities [{id(null), begin, end}, ...]
});

8. AdEvent

player.addEventListener(AdaptivePlayer.EventType.AdEvent, function (e) {
    if (e.data.type == AdaptivePlayer.AD.START_EVENT) {
        // AD start
        e.data.urls.forEach(function (item) {
            if (item.name = AdaptivePlayer.AD.DEFAULT_CLICK) {
                // item.url: defaultClick url
            } else if (item.name = AdaptivePlayer.AD.TRACK_CLICK) {
                // item.url: trackClick url
            }
        })
    } else if (e.data.type == AdaptivePlayer.AD.END_EVENT) {
        // AD end
    }
});
