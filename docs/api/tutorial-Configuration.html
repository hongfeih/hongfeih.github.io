Tutorial: Configuration

Configuration

Tips

  1. Should be called before load
  2. Default value:
    {
 withCredentialsFragmentPattern: [
   '/nlds/', // neulion live
   '/nldsu/',
   '/nlds_vod/', // neulion vod
   '/nldsu_vod/'
 ],
 streaming: {
   neulionGapDealing: true,
   safeEdgeEndSize: 0.10,
   safeEdgeBeginSize: 0.02,
   ignoreTextStreamFailures: true,
   jumpLargeGaps: true
 },
 preferredAudioLanguage: "",
 preferredTextLanguage: "", 
 useExternalM3U8Parser: true,
 appendXPLAYForStitching: false,
 dashCC: true,
 closedCaption: {
   label: 'English(CC)',
   language: 'en'
 }
}

Example

    var config = {
        useExternalM3U8Parser: false, // UTC timing for Safari, bydefault is true
        abr: {
            enabled: true, // auto bitrate control
            startBitrate: 1500000 //default bitrate
        },
        streaming: {
            bufferingGoal: 10, //The number of seconds of content before playback, 10 by default. 
        },
        drm: {
            widevine: {
              server: licenseServer_WV, // licenser server url of widevine
            },
            playready: {
              server: licenseServer_PR, // licenser server url of playready
            },
            fairplay: {
              server: licenseServer_FP, // licenser server url of fairplay
              serverCertificate: certUrl // certificate file of fairplay, may be changed later
            },
            params: {
              token: cro_token, // jwt token from app server
            }
        },
        withCredentials: withCredentials,
        preferredAudioLanguage: "fra", // set default language
        preferredTextLanguage: "fra",
        closedCaption: { // set default CC language
            label: 'English',
            language: 'en'
        }
        //headers: { 'key1':'data1', 'key2': 'data2' } //customized headers when needed
    };
    player.configure(config)
