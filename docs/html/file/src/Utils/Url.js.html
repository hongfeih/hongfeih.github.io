    import UrlValidator from '../../externals/valid-url/index'
import { AdaptivePlayer } from '../AdaptivePlayer';

export function validateURL(url) {
  return UrlValidator.isWebUri(url);
}

export function isAbsoluteUrl(url) {
  if (typeof url !== 'string') {
    return false;
  }
  return UrlValidator.isWebUri(url);
}

export function getPath(url) {
  let tmp = url.split('?')[0];
  return tmp.substr(0, tmp.lastIndexOf('/') + 1);
}

export function getExtension(url) {
  if (!url) { return ''; }

  let parts = url.split('?')[0].split('.');
  // Handle files with no extensions and hidden files with no extension
  if (parts.length === 1 || (parts[0] === '' && parts.length === 2)) {
    return '';
  }
  return parts.pop().toLowerCase();
}

export function detectPlayerType(url, protocol, isDRM) {
  if (url == null || url === '' || !validateURL(url)) {
    return AdaptivePlayer.PlayerType.UNKNOWN;
  }

  if (typeof protocol !== 'undefined') {
    switch (protocol) {
      case AdaptivePlayer.Protocol.HTTP:
        return AdaptivePlayer.PlayerType.NATIVE;
      case AdaptivePlayer.Protocol.WEBRTC:
        return AdaptivePlayer.PlayerType.WEBRTC;
      case AdaptivePlayer.Protocol.DASH:
        return AdaptivePlayer.PlayerType.SHAKA;
      case AdaptivePlayer.Protocol.HLS:
        return (AdaptivePlayer.browserType === 'Safari' || AdaptivePlayer.browserType === 'Mobile Safari' /* || browserType === 'Edge' */)
          ? AdaptivePlayer.PlayerType.NATIVE : AdaptivePlayer.PlayerType.HLSJS;
      default:
        return AdaptivePlayer.PlayerType.UNKNOWN;
    }
  } else {
    if (url.indexOf('ws://') !== -1 || url.indexOf('wss://') !== -1) { // webrtc streams
      return AdaptivePlayer.PlayerType.WEBRTC;
    }
    let ext = getExtension(url);
    if (ext === 'm3u8' || ext === 'm3u') {
      // Safari, use hls.js for non-drm contents
      // Or use native player for DRM contents
      // Mobile Safari uses native player (DRM not supported)
      // Other browser use hls.js
      if ((!!isDRM && AdaptivePlayer.browserType === 'Safari') || AdaptivePlayer.browserType === 'Mobile Safari' /* || browserType === 'Edge' */) {
        return AdaptivePlayer.PlayerType.NATIVE;
      }
      return AdaptivePlayer.PlayerType.HLSJS;
    } else if (ext === 'mpd') {
      return AdaptivePlayer.PlayerType.SHAKA;
    } else if (ext === 'mp4' || ext === 'webm' || ext === 'ogg') {
      return AdaptivePlayer.PlayerType.NATIVE;
    } else {
      return AdaptivePlayer.PlayerType.UNKNOWN;
    }
  }
}
