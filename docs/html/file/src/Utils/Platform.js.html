    import UAParser from '../../externals/ua-parser-js/ua-parser.min'

export function probePlatformSupport() {
  window.MediaSource = window.MediaSource || window.WebKitMediaSource;
  let mseSupport = !!(window.MediaSource &&
    typeof window.MediaSource.isTypeSupported === 'function' &&
    window.MediaSource.isTypeSupported('video/mp4; codecs="avc1.42E01E,mp4a.40.2"'));

  let browser = new UAParser().getBrowser();
  let browserLimitation = (browser.name !== 'IE') || (browser.name === 'IE' && browser.major >= 11); // IE 11+

  return mseSupport && browserLimitation;
}

export function probeWebRTCSupport() {
  let isWebRTCSupported = false;
  ['RTCPeerConnection', 'webkitRTCPeerConnection', 'mozRTCPeerConnection', 'RTCIceGatherer'].forEach(function (item) {
    if (isWebRTCSupported) {
      return;
    }

    if (item in window) {
      isWebRTCSupported = true;
    }
  });
  return isWebRTCSupported;
}

function isLocalStorageAvailable() {
  let test = 'test';
  try {
    localStorage.setItem(test, test);
    localStorage.removeItem(test);
    return true;
  } catch (e) {
    return false;
  }
}

export function generateUUID() {
  // http://stackoverflow.com/questions/105034/create-guid-uuid-in-javascript
  return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function (c) {
    let r = Math.random() * 16 | 0;
    let v = c === 'x' ? r : r & 0x3 | 0x8;
    return v.toString(16);
  });
}

export function getDeviceId() {
  let id = generateUUID();

  // detect local storage available or not
  // if not available, use different id every time
  if (isLocalStorageAvailable()) {
    let oldId = localStorage.getItem('deviceId');
    if (!oldId) {
      localStorage.setItem('deviceId', id);
    } else {
      id = oldId;
    }
  }

  return id;
}
