    export function mergeObject(obj1, obj2) {
  let obj3 = {};
  for (let attrname in obj1) { obj3[attrname] = obj1[attrname]; }
  for (let attrname in obj2) { if (obj3[attrname] == null) obj3[attrname] = obj2[attrname]; }
  return obj3;
}

// Speed up calls to hasOwnProperty
let hasOwnProperty = Object.prototype.hasOwnProperty;
export function isEmpty(obj) {
  // null and undefined are 'empty'
  if (obj == null) { return true; }

  // Assume if it has a length property with a non-zero value
  // that that property is correct.
  if (obj.length > 0) { return false; }
  if (obj.length === 0) { return true; }

  // If it isn't an object at this point
  // it is empty, but it can't be anything *but* empty
  // Is it empty?  Depends on your application.
  if (typeof obj !== 'object') { return true; }

  // Otherwise, does it have any properties of its own?
  // Note that this doesn't handle
  // toString and valueOf enumeration bugs in IE < 9
  for (let key in obj) {
    if (hasOwnProperty.call(obj, key)) { return false; }
  }

  return true;
}

export function getKeyByValue(map, value) {
  for (let name in map) {
    if (map[name] === value) {
      return name;
    }
  }
  return '';
}

export function find(arr, propertyName, target) {
  if (!arr || arr.length <= 0) {
    return null;
  }

  let n = arr.length;
  for (let i = 0; i < n; i++) {
    if (arr[i][propertyName] === target) {
      return arr[i];
    }
  }

  return null;
}
