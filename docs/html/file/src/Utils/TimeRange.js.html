export function absTimeToRelTime(time, timeRanges, absStart) {
  if (time < absStart) {
    time = absStart;
  } else if (timeRanges.length > 0 && time > timeRanges[timeRanges.length - 1].end) {
    time = timeRanges[timeRanges.length - 1].end;
  }

  let calcTime = time;
  let offset = 0;
  for (let i = 0; i < timeRanges.length; i++) {
    if (i !== 0) {
      offset += (timeRanges[i].begin - timeRanges[i - 1].end);
    }
    if (time >= timeRanges[i].begin && time <= timeRanges[i].end) {
      calcTime = time - offset - absStart;
      break;
    } else {
      if (time < timeRanges[i].begin) {
        calcTime = timeRanges[i].begin - offset - absStart;
        break;
      }
      if (i !== 0) {
        if (time > timeRanges[i - 1].end && time < timeRanges[i].begin) {
          calcTime = timeRanges[i].begin - offset - absStart;
          break;
        }
        if (i === timeRanges.length - 1 && time > timeRanges[i].end) {
          calcTime = timeRanges[i].end - offset - absStart;
          break;
        }
      }
    }
  }
  return calcTime;
}

export function relTimeToAbsTime(time, timeRanges, absStart) {
  let calcTime = time;
  let offset = 0;
  for (let i = 0; i < timeRanges.length; i++) {
    if (i !== 0) {
      offset += (timeRanges[i].begin - timeRanges[i - 1].end);
    }
    let relBegin = timeRanges[i].begin - offset - absStart;
    let relEnd = timeRanges[i].end - offset - absStart;
    if (time >= relBegin && time <= relEnd) {
      calcTime = time + offset + absStart;
      break;
    } else {
      if (time < relBegin) {
        calcTime = relBegin + offset + absStart;
        break;
      }
      if (i !== 0) {
        let lastOffset = offset - (timeRanges[i].begin - timeRanges[i - 1].end);
        let lastRelEnd = timeRanges[i - 1].end - lastOffset - absStart;
        if (time > lastRelEnd && time < relBegin) {
          calcTime = relBegin + offset + absStart;
          break;
        }
        if (i === timeRanges.length - 1 && time > relEnd) {
          calcTime = relEnd + offset + absStart;
          break;
        }
      }
    }
  }
  return calcTime;
}

export function getAbsSafeTime(absTime, timeRanges, absEndLimit) {
  let safeAbsTime = absTime;
  if (timeRanges && timeRanges.length > 0) {
    if (absTime <= timeRanges[0].begin) {
      safeAbsTime = timeRanges[0].begin;
      return safeAbsTime;
    }
    if (absTime >= timeRanges[timeRanges.length - 1].end) {
      safeAbsTime = timeRanges[timeRanges.length - 1].end;
      if (absEndLimit) {
        safeAbsTime = Math.min(absEndLimit, safeAbsTime);
      }
      return safeAbsTime;
    }
    for (let i = 0; i < timeRanges.length; i++) {
      let lastEnd = timeRanges[i].begin;
      if (i !== 0) {
        lastEnd = timeRanges[i - 1].end;
      }
      if (absTime >= lastEnd && absTime <= timeRanges[i].begin) {
        safeAbsTime = timeRanges[i].begin;
        break;
      }
      if (absTime > timeRanges[i].begin && absTime <= timeRanges[i].end) {
        break;
      }
    }
  }

  return safeAbsTime;
}

export function vacancyTime(absTime, timeRanges) {
  let vacancy = true;

  for (let i = 0; i < timeRanges.length; i++) {
    if (absTime >= timeRanges[i].begin && absTime <= timeRanges[i].end) {
      vacancy = false;
      break;
    }
  }

  return vacancy;
}

function _findTimeRanges(timeRanges, time) {
    let range = null;
    for (let i = timeRanges.length - 1; i >= 0; i--) {
        if ((timeRanges[i].end === null && time >= timeRanges[i].begin) ||
            (timeRanges[i].end !== null && time >= timeRanges[i].begin && time <= timeRanges[i].end)) {
            range = timeRanges[i];
            break;
        }
    }
    return range;
}

export function mergeProgrameTime(timeline, dateRangesInM3u, targetDurationInM3u, playlistDurationInM3u) {
  let dateRanges = dateRangesInM3u;
  let timeRanges = timeline.timeRanges;

  timeline.targetDuration = targetDurationInM3u || 0;

  if (dateRanges.length > 0) {
    for (let i = 0; i < dateRanges.length; i++) {
      if (i === 0 && !timeline.init) {
        timeline.init = true;
        timeline.absStart = dateRanges[i].begin;
      }
      let range = _findTimeRanges(timeline.timeRanges, dateRanges[i].begin);
      if (range === null) {
        range = {
          begin: dateRanges[i].begin,
          end: dateRanges[i].end
        };
        timeRanges.push(range);
      } else {
        if (range.end < dateRanges[i].end) {
            range.end = dateRanges[i].end;
        }
      }
      if (i === 0) {
        timeline.curAbsStart = dateRanges[i].begin;
      }
      if (i === dateRanges.length - 1) {
        timeline.curAbsEnd = dateRanges[dateRanges.length - 1].end;
      }
    }
  } else {
    timeline.curAbsEnd = playlistDurationInM3u;
  }
}
