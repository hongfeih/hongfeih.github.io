    import { AdaptivePlayer } from '../AdaptivePlayer';
import { ID3 } from '../Media/ID3'
import { FakeEvent } from '../Utils/FakeEvent'
import { Logger } from '../Utils/Logger';

export class CuepointManager extends Logger {
  constructor(eventHandler) {
    super(eventHandler._logger);

    this._eventHandler = eventHandler;

    this._cuepoints = [];
    this._cuepoints_bak = []; // backup of cuepoints
  }

  dispatch(pts, cue) {
    this._eventHandler.dispatchEvent(new FakeEvent(AdaptivePlayer.EventType.ID3Event, { start: pts, end: pts, value: cue }));
    this.print(AdaptivePlayer.LogLevel.DEBUG, 'CuePointManager', 'CuePoint Triggered: ' + pts);

    if (cue && cue.dict && cue.dict.cueType && (cue.dict.cueType === 'start' || cue.dict.cueType === 'keep')) {
      // clear wrong cues
      this._eventHandler.clearWrongCues();
    }
  }

  cleanup() {
    this._cuepoints.splice(0, this._cuepoints.length);
    this._cuepoints_bak.splice(0, this._cuepoints_bak.length);
  }

  clearID3() {
    // restore cuepoints with backup after seek
    this._cuepoints.splice(0, this._cuepoints.length);
    this._cuepoints = this._cuepoints_bak.slice();
  }

  onTimeUpdate(event) {
    // process cuepoints
    this._cuepoints.sort(function (a, b) {
      return a.pts - b.pts;
    });
    for (let index = 0; index < this._cuepoints.length; index++) {
      let sample = this._cuepoints[index];
      if (sample.pts <= event.currentTime && sample.pts >= event.currentTime - 1) {
        let cue = {
          key: sample.dict['key'],
          dict: sample.dict
        };

        // remove it
        this._cuepoints.splice(index, 1);
        index--;

        // should send event after remove
        this.dispatch(sample.pts, cue);
      } else if (sample.pts > event.currentTime) {
        break;
      }
    }
  }

  onID3(sample) {
    if (!sample || !(sample.data || sample.unit) || isNaN(sample.pts)) {
      this.print(AdaptivePlayer.LogLevel.WARNING, 'CuePointManager', 'onID3: Wrong ID3 data');
      return;
    }

    // remove item with same pts
    if (this._cuepoints.some(function (item) {
      return item.pts === sample.pts && item.type === CuepointManager.CUE_TYPE.ID3.value; 
    })) {
      this.print(AdaptivePlayer.LogLevel.WARNING, 'CuePointManager', 'onID3: ID3 with sample pts already exists: ' + sample.pts);
      return;
    }

    let data = sample.unit || sample.data;
    let id3 = new ID3(data);

    sample.dict = {};
    let id3Dict = id3.dict();
    for (let keyIndex in id3Dict) {
      sample.dict[keyIndex] = id3Dict[keyIndex];
    }

    sample.type = CuepointManager.CUE_TYPE.ID3.value;
    if (typeof sample.dict.type !== 'undefined') {
      sample.dict['tag'] = sample.dict['type'];
    }
    sample.dict['type'] = CuepointManager.CUE_TYPE.ID3.name;
    this._cuepoints.push(sample);
    this._cuepoints_bak.push(sample);
    // sort by time
    this._cuepoints.sort(function (a, b) {
      return a.pts - b.pts;
    });
    this._cuepoints_bak.sort(function (a, b) {
      return a.pts - b.pts;
    });
  }

  onNativeID3(cue) {
    try {
      let cueData = {
        raw: cue.value || cue.data,
        key: cue.value.key,
        dict: cue.value['data']['ad-list'] ? cue.value['data']['ad-list'][0] : cue.value
      };
      cueData.dict['type'] = CuepointManager.CUE_TYPE.ID3.name;
      if (cueData.dict.data) {
        cueData.dict['value'] = cueData.dict.data;
        if (!cueData.dict.info) {
          cueData.dict.info = cueData.dict.data;
        }
      }
      this.print(AdaptivePlayer.LogLevel.TRACE, 'CuePointManager', 'Native ID3 got: ' + cue.startTime);
      this.dispatch(cue.startTime, cueData);
    } catch (e) {
      this.print(AdaptivePlayer.LogLevel.WARNING, 'CuePointManager', 'Native ID3 got, but no adlist value');
    }
  }

  addCuepoints(cues) {
    cues.forEach(function (element) {
      if (!element || !(element.dict) || !(element.time)) {
        return;
      }
      element.pts = element.time;
      element.data = {};
      element.type = CuepointManager.CUE_TYPE.CUE_POINT.value;
      element.dict['type'] = CuepointManager.CUE_TYPE.CUE_POINT.name;
      this._cuepoints.push(element);
      this._cuepoints_bak.push(element);
    }, this);

    // sort by time
    this._cuepoints.sort(function (a, b) { return a.pts - b.pts; });
    this._cuepoints_bak.sort(function (a, b) { return a.pts - b.pts; });

    return true;
  }

  removeCuepoint(time, name) {
    let ret = false;
    if (!(time) || !(name)) {
      return ret;
    }
    [this._cuepoints, this._cuepoints_bak].forEach(function (array) {
      for (let i = 0; i < array.length; i++) {
        let sample = array[i];
        if (sample && sample.type === CuepointManager.CUE_TYPE.CUE_POINT.value && 
          (sample.pts && sample.pts === time) && 
          (sample.dict && sample.dict.name && sample.dict.name === name)) {
          array.splice(i, 1);
          i--;
          ret = true;
        }
      }
    });
    return ret;
  }

  getCuepoints() {
    let ret = [];
    for (let i = 0; i < this._cuepoints_bak.length; i++) {
      let sample = this._cuepoints_bak[i];
      if (sample && sample.type === CuepointManager.CUE_TYPE.CUE_POINT.value) {
        ret.push(sample);
      }
    }
    return ret;
  }

  removeAllCuepoints() {
    [this._cuepoints, this._cuepoints_bak].forEach(function (array) {
      for (let i = 0; i < array.length; i++) {
        let sample = array[i];
        if (sample && sample.type === CuepointManager.CUE_TYPE.CUE_POINT.value) {
          array.splice(i, 1);
          i--;
        }
      }
    });
  }
}

CuepointManager.CUE_TYPE = {
  ID3: { value: 1, name: 'id3' },
  CUE_POINT: { value: 2, name: 'cuepoint' }
}
