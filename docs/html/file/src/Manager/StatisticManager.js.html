    import { AdaptivePlayer } from '../AdaptivePlayer';
import { NLError } from '../Media/NLError'
import { FakeEvent } from '../Utils/FakeEvent';
import { Logger } from '../Utils/Logger'

export class StatisticManager extends Logger {
  constructor(video, parent) {
    super(parent._logger);

    this._eventHandler = parent;
    this._video = video;
    this.startWatchDog = null;
    this.seekWatchDog = null;

    this._onError = null;
    this._errConfig = {
      url: '',
      level: 0
    };
    this.errorCheckList = [];

    StatisticManager.WATCHDOG_TIMEOUT = 10 * 1000;
    StatisticManager.ERROR_CHECK_INTERVAL = 30 * 1000;
  }

  stop() {
    this.stopLoadWatchDog();
    this.stopSeekWatchDog();
  }

  setErrorCallback(onError, errorConfig) {
    this._onError = onError;
    this._errConfig = errorConfig;
  }

  getErrorConfig() {
    return this._errConfig;
  }

  // error in [ 30s ] error don't add
  _generateCheckFunction(olderErrorInfo) {
    return function (olderErrorInfo, newerErrorInfo) {
      return arguments[0].compare(arguments[1]);
    }.bind(null, olderErrorInfo);
  }

  // return true when same mesage in 30s
  _onCheckErrorInfo(addedErrorInfo) {
    for (let i = 0; i < this.errorCheckList.length; i++) {
      let errorItem = this.errorCheckList[i];
      if (Date.now() - errorItem.time > StatisticManager.ERROR_CHECK_INTERVAL) {
        this.errorCheckList.splice(i, 1);
        i--;
        continue;
      }
      if (errorItem.checkFunc(addedErrorInfo)) {
        return true;
      }
    }

    this.errorCheckList.push({
      checkFunc: this._generateCheckFunction(addedErrorInfo),
      time: Date.now()
    });
    return false;
  }

  reset(url) {
    if (this.startWatchDog) {
      this.stopLoadWatchDog();
      this.startWatchDog = null;
    }
    
    if (this.seekWatchDog) {
      this.stopSeekWatchDog();
      this.seekWatchDog = null;
    }
    
    this.errorCheckList = [];
    this._errConfig.url = url;
    this.startLoadWatchDog();
  }

  startLoadWatchDog() {
    if (this.startWatchDog) {
      this.stopLoadWatchDog();
    }
    this.startWatchDog = window.setTimeout(function () {
      let message = 'Start playback timeout';
      let code = NLError.Code.Start_Playback_Timeout;
      let level = NLError.SeverityLevel.Warning;
      let error = NLError.create(code, level, this._errConfig.url, message);
      this.reportError(error);
    }.bind(this), StatisticManager.WATCHDOG_TIMEOUT);
  }

  startSeekWatchDog() {
    if (this.seekWatchDog) {
      this.stopSeekWatchDog();
    }
    this.seekWatchDog = window.setTimeout(function () {
      let message = 'Seek timeout';
      let code = NLError.Code.Seek_Timeout;
      let level = NLError.SeverityLevel.Warning;
      let error = NLError.create(code, level, this._errConfig.url, message);
      this.reportError(error);
    }.bind(this), StatisticManager.WATCHDOG_TIMEOUT);
  }

  stopLoadWatchDog() {
    if (this.startWatchDog) {
      this.startWatchDog = window.clearTimeout(this.startWatchDog);
    }
  }

  stopSeekWatchDog() {
    if (this.seekWatchDog) {
      this.seekWatchDog = window.clearTimeout(this.seekWatchDog);
    }
  }

  reportUnsupportedSchema(url) {
    let message = 'load with wrong url!';
    let code = NLError.Code.Get_Primary_Manifest_Failed + NLError.NetworkDetailCode.Connect_Failed;
    let level = NLError.SeverityLevel.Fatal;
    this.reportError(NLError.create(code, level, url, message));
  }

  reportUnsupported() {
    let message = 'Your Browser does not support MSE or WebRTC';
    let code = NLError.Code.Platform_Not_Supported;
    let level = NLError.SeverityLevel.Fatal;
    let error = NLError.create(code, level, this._errConfig.url, message);
    this.reportError(error);
  }

  reportNativeVideoError(msg, errorCode) {
    let code = errorCode || NLError.Code.Playback_Generic_Error;
    let level = NLError.SeverityLevel.Fatal;
    let error = NLError.create(code, level, this._errConfig.url, msg);
    this.reportError(error);
  }

  reportError(error) {
    if (!error.isValid() ||
      this._onCheckErrorInfo(error) ||
      (this._errConfig && this._errConfig.level && this._errConfig.level < error.level)) {
      return;
    }

    if (this._onError && error.level <= NLError.SeverityLevel.Error) {
      this._onError(error.toJSON());
    }

    if (error.level > NLError.SeverityLevel.Error) {
      error.code += 5000;
    }
    
    this._eventHandler.dispatchEvent(new FakeEvent(AdaptivePlayer.EventType.Error, { details: error.toJSON() }));
    this.print(AdaptivePlayer.LogLevel.INFO, 'StatisticManager', 'Trigger Error event: ' + JSON.stringify(error.toJSON()));
  }
}
