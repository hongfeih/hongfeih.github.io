    import * as UrlUtil from '../Utils/Url'
import * as XHRUtil from '../Utils/Xhr'
import { Parser } from '../../3rd/m3u8-parser/dist/m3u8-parser.min'
import { NLError } from './NLError'
import { WebVTTParser } from '../../externals/webvtt/parser'

export default class HlsPlaylistParser {
  constructor() {
    this._playHeadTimeCb = null;
    this._errorCb = null;
    this._thumbnailCb = null;
    this._thumbnailImageCb = null;
    this._adCb = null;
    this._masterUrl = null;
    this._baseUrl = null;
    this._levelUrl = null;
    this._levelRedundantUrls = [];
    this._levelIndex = -1;
    this._ccInfos = null;
  }

  destroy() {
    this._playHeadTimeCb = null;
    this._errorCb = null;
    this._thumbnailCb = null;
    this._thumbnailImageCb = null;
    this._adCb = null;
    this._masterUrl = null;
    this._baseUrl = null;
    this._levelUrl = null;
    this._levelRedundantUrls = [];
    this._levelIndex = -1;
    this._ccInfos = null;
  }

  loadPlaylist(url, userdata) {
    if (this._masterUrl == null) {
      this._masterUrl = url;
      this._baseUrl = UrlUtil.getPath(url);
    }
    let async = XHRUtil.createXHREx({
      url: url
    }, userdata).then(function (ab) {
      let parser = new Parser();
      parser.push(ab.response);
      parser.end();

      if (ab.userdata && ab.userdata.type === 'master' && parser.isMasterPlaylist()) {
        try {
          this._parseMasterPlaylist(parser);
        } catch (e) {
          this._onError(NLError.Basic.M3U8.MANIFEST_PARSING_ERROR, e);
        }
      }

      if (ab.userdata && (ab.userdata.type === 'level' || ab.userdata.type === 'master') && parser.isLevelPlaylist()) {
        if (ab.userdata.type === 'master') {
          this._levelUrl = this._masterUrl;
        }
        try {
          this._parseLevelPlaylist(parser, ab.userdata.index);
        } catch (e) {
          if (ab.userdata.type === 'level' && this._levelIndex >= 0 && this._levelIndex < this._levelRedundantUrls.length) {
            this._levelIndex += 1;
            this._levelUrl = this._levelRedundantUrls[this._levelIndex];
          } else {
            this._onError(NLError.Basic.M3U8.GET_PLAYHEADTIME_FAILED, e);
          }
        }
      }

      if (ab.userdata && ab.userdata.type === 'thumbnail' && parser.isLevelPlaylist()) {
        try {
          this._parseThumbnailPlaylist(parser, ab.userdata);
        } catch (e) {
          this._onError(NLError.Basic.M3U8.GET_THUMBNAIL_FAILED, e);
        }
      }
    }.bind(this)).catch(function (e) {
      this._onError(NLError.Basic.M3U8.MANIFEST_LOAD_ERROR, e.response);
    }.bind(this));

    return async;
  }

  loadVtts(url, userdata) {
    let async = XHRUtil.createXHREx({
      url: url
    }, userdata).then(function (ab) {
      // parse vtts
      if (userdata && ab.userdata && ab.userdata.type === 'vtts') {
        let parser = new WebVTTParser();
        let result = parser.parse(ab.response, 'metadata');
        try {
          this._parseVttPlaylist(result, ab.userdata);
        } catch (e) {
          this._onError(NLError.Basic.M3U8.PARSE_VTT_FAILED, e);
        }
      }
    }.bind(this)).catch(function (e) {
      this._onError(NLError.Basic.M3U8.GET_VTT_FAILED, e.response);
    }.bind(this));

    return async;
  }

  setPlayHeadCB(playHeadTimeCb) {
    this._playHeadTimeCb = playHeadTimeCb;
  }

  setErrorCb(errorCb) {
    this._errorCb = errorCb;
  }

  setThumbnailCb(thumbnailCb) {
    this._thumbnailCb = thumbnailCb;
  }

  setThumbnailImageCb(thumbnailImageCb) {
    this._thumbnailImageCb = thumbnailImageCb;
  }

  setAdStitchCb(adCb) {
    this._adCb = adCb;
  }

  getAbsUrl(url) {
    let absUrl = url;
    if (!UrlUtil.isAbsoluteUrl(url) && this._baseUrl) {
      absUrl = this._baseUrl + url;
    }
    return absUrl;
  }

  getLevelUrl() {
    return this._levelUrl;
  }

  getHlsCC() {
    return this._ccInfos;
  }

  _onError(descript, error) {
    if (this._errorCb) {
      this._errorCb({ error: descript, code: error });
    }
  }

  _parseMasterPlaylist(parser) {
    if (parser.manifest && parser.manifest.mediaGroups && parser.manifest.mediaGroups['CLOSED-CAPTIONS'] && 
        parser.manifest.mediaGroups['CLOSED-CAPTIONS'].ccs) {
      let ccs = parser.manifest.mediaGroups['CLOSED-CAPTIONS'].ccs;
      let index = 1;
      if (ccs && Object.keys(ccs).length > 0) {
        this._ccInfos = {};
        for (let i in ccs) {
          this._ccInfos['cc' + index.toString()] = {
            name: i,
            lang: ccs[i].language
          };
        }
      }
    }
    for (let i = 0; i < parser.manifest.playlists.length; i++) {
      let sbrUrl = parser.manifest.playlists[i].uri;
      if (!UrlUtil.isAbsoluteUrl(sbrUrl)) {
        sbrUrl = this._baseUrl + sbrUrl;
      }
      if (this._levelUrl == null) {
        this._levelUrl = sbrUrl;
        this._levelIndex = i;
      }
      this._levelRedundantUrls.push(sbrUrl);
    }
    this.loadPlaylist(this._levelUrl, { type: 'level', index: 0 });

    let thumbnails = parser.getThumbnailsStreams();
    if (thumbnails.length > 0 && this._thumbnailCb) {
      let upStreams = [];
      for (let i = 0; i < thumbnails.length; i++) {
        let thumbnailUri = thumbnails[0].uri;
        if (!UrlUtil.isAbsoluteUrl(thumbnailUri)) {
          thumbnailUri = this._baseUrl + thumbnailUri;
        }
        // this.loadPlaylist(thumbnailUri, {type: 'thumbnail', index: i});
        let upStream = {
          id: i,
          width: thumbnails[0].width,
          height: thumbnails[0].height,
          uri: thumbnailUri
        };
        upStreams.push(upStream);
      }
      this._thumbnailCb(upStreams);
    }
  }

  _parseLevelPlaylist(parser, index) {
    if (this._playHeadTimeCb) {
      this._playHeadTimeCb(parser.manifest, index);
    }
    if (this._adCb) {
      this._adCb(parser.adsInfos);
    }
  }

  _parseThumbnailPlaylist(parser, userdata) {
    if (this._thumbnailCb) {
      this._thumbnailCb([], parser, userdata);
    }
  }

  _parseVttPlaylist(result, userdata) {
    let index = userdata.index;
    let time = userdata.time;
    let baseTime = userdata.baseTime;
    let requestTime = userdata.requestTime;

    let thumbnails = {
      'images': [],
      'baseTime': baseTime,
      'requestTime': time,
      'streamId': index
    };
    if (this._thumbnailImageCb) {
      if (result.cues && result.cues.length > 0) {
        for (let i = 0; i < result.cues.length; i++) {
          let parts = result.cues[i].text.split('#xywh=');
          if (parts != null && parts.length === 2) {
            let positions = parts[1].split(',');
            if (positions == null || positions.length !== 4) {
              continue;
            }
            let image = {
              'url': this.getAbsUrl(parts[0]),
              'startTime': baseTime + result.cues[i].startTime,
              'endTime': baseTime + result.cues[i].endTime,
              'topX': positions[0],
              'topY': positions[1],
              'width': positions[2],
              'height': positions[3]
            };
            thumbnails.images.push(image);
          }
        }
        this._thumbnailImageCb(thumbnails, requestTime);
      }
    }
  }
}
