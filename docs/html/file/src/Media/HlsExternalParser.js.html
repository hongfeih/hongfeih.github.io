    import * as UrlUtil from '../Utils/Url'
import * as XHRUtil from '../Utils/Xhr'
import m3u8Parser from 'm3u8-parser'
import { NLError } from './Error'

export function getPlayHeadTime(url, callback, onError) {
  _processPlayHeadTime(url, callback, onError);
};

function _processPlayHeadTime(url, callback, onError) {
  var baseUrl = UrlUtil.getPath(url);
  var async = XHRUtil.createXHR({
    url: url
  }).then(function (ab) {
    var parser = new m3u8Parser.Parser();
    parser.push(ab);
    parser.end();
    if (parser.manifest && parser.manifest.playlists && parser.manifest.playlists.length > 0) {
      // Master;
      var sbrUrl = parser.manifest.playlists[0].uri;
      if (!UrlUtil.isAbsoluteUrl(sbrUrl))
        sbrUrl = baseUrl + sbrUrl;
      _processPlayHeadTime(sbrUrl, callback, onError); // download and parse SBR m3u8
    } else if (parser.manifest && parser.manifest.segments && parser.manifest.segments.length > 0) {
      // SBR
      callback(parser.manifest);
    } else {
      onError({ error: NLError.Basic.M3U8.MANIFEST_PARSING_ERROR });
    }
  }.bind(this)).catch(function (e) {
    onError({ error: NLError.Basic.M3U8.GET_PLAYHEADTIME_FAILED, code: e });
  }.bind(this));

  return async;
};
