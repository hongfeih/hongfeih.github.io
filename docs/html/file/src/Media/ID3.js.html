import * as StringUtil from '../Utils/String'

export class ID3 {
    constructor(data) {
        this._dict = {};
        this._hasTimeStamp = false;
        let offset = 0;
        let byte1;
        let byte2;
        let byte3;
        let byte4;
        let tagSize;
        let endPos;
        let header;
        let len;
        do {
            header = this._readUTF(data, offset, 3);
            offset += 3;
            // first check for ID3 header
            if (header === 'ID3') {
                // skip 24 bits
                offset += 3;
                // retrieve tag(s) length
                byte1 = data[offset++] & 0x7f;
                byte2 = data[offset++] & 0x7f;
                byte3 = data[offset++] & 0x7f;
                byte4 = data[offset++] & 0x7f;
                tagSize = (byte1 << 21) + (byte2 << 14) + (byte3 << 7) + byte4;
                endPos = offset + tagSize;

                // read ID3 tags
                this._parseID3Frames(data, offset, endPos);
                offset = endPos;
            } else if (header === '3DI') {
                // http://id3.org/id3v2.4.0-structure chapter 3.4.   ID3v2 footer
                offset += 7;
                offset = endPos;
            } else {
                offset -= 3;
                len = offset;
                if (len) {
                    if (!this._hasTimeStamp) {
                        // ID3 tag found, but no timestamp
                    }
                    this._length = len;
                    this._payload = data.subarray(0, len);
                }
                return;
            }
        } while (offset < endPos);
    }

    _readUTF(data, start, len) {
        let result = '';
        let offset = start;
        let end = start + len;
        do {
            result += String.fromCharCode(data[offset++]);
        } while (offset < end);
        return result;
    }

    _getFrameSize(data, offset) {
        let size = 0;
        size = ((data[offset] & 0x7f) << 21);
        size |= ((data[offset + 1] & 0x7f) << 14);
        size |= ((data[offset + 2] & 0x7f) << 7);
        size |= (data[offset + 3] & 0x7f);
        return size;
    }

    _decodeTextFrame(frame, offset, type, size) {
        if (size < 2) {
            return null;
        }

        if (type === 'TXXX') {
            /*
            Format:
            [0]   = {Text Encoding}
            [1-?] = {Description}\0{Value}
            */
            let index = 1;
            let description = StringUtil.utf8ArrayToStr(frame.subarray(offset + index));

            index += description.length + 1;
            let value = StringUtil.utf8ArrayToStr(frame.subarray(offset + index));

            return { info: description, data: value };
        } else {
            /*
            Format:
            [0]   = {Text Encoding}
            [1-?] = {Value}
            */
            let text = StringUtil.utf8ArrayToStr(frame.subarray(offset + 1));
            return { info: '', data: text };
        }
    }

    _decodeURLFrame(frame, offset, type, size) {
        if (type === 'WXXX') {
            /*
            Format:
            [0]   = {Text Encoding}
            [1-?] = {Description}\0{URL}
            */
            if (size < 2) {
                return null;
            }

            let index = 1;
            let description = StringUtil.utf8ArrayToStr(frame.subarray(offset + index));

            index += description.length + 1;
            let value = StringUtil.utf8ArrayToStr(frame.subarray(offset + index));

            return { info: description, data: value };
        } else {
            /*
            Format:
            [0-?] = {URL}
            */
            let url = StringUtil.utf8ArrayToStr(frame.data);
            return { info: '', data: url };
        }
    }

    _parseID3Frames(data, offset, endPos) {
        let tagId;
        // let tagStart;
        // let tagLen;
        // let tagFlags;
        let timestamp;
        if (offset + 8 <= endPos) {
            tagId = this._readUTF(data, offset, 4);
            offset += 4;
            this._dict['key'] = tagId;
            switch (tagId) {
                case 'GEOB':
                    // skip tag flags and len;
                    offset += 7;
                    if (this._readUTF(data, offset, 15) === 'application/xml') {
                        offset += 15;

                        offset += 1;
                        if (this._readUTF(data, offset, 18) === 'substitution.plist') {
                            offset += 18;

                            offset += 2;

                            let xmlStr = this._readUTF(data, offset, endPos - offset);
                            offset += endPos - offset;
                            let parser = new DOMParser();
                            let xmlDoc = parser.parseFromString(xmlStr, 'application/xml');
                            if (xmlDoc.getElementsByTagName('dict').length > 1) {
                                // assert(xmlDoc.getElementsByTagName('dict')[0].childNodes[0].nodeValue === 'ad-list');
                                let nodes = xmlDoc.getElementsByTagName('dict')[1].childNodes;
                                let nodeValues = [];
                                for (let i = 0; i < nodes.length; i++) {
                                    if (nodes[i].nodeType === 1 && nodes[i].nodeName) { // element type and had the node name
                                        nodeValues.push({ node: nodes[i].nodeName, value: nodes[i].textContent.trim() });
                                    }
                                }
                                // assert(nodeValues.length % 2 === 0);
                                for (let j = 0; j < nodeValues.length; j++) {
                                    let keyIndex = j++;
                                    // assert(nodeValues[keyIndex].node === 'keyIndex');
                                    this._dict[nodeValues[keyIndex].value] = nodeValues[j].value;
                                }
                            }
                        }
                    }
                    break;
                case 'PRIV':
                    // tagLen = data[offset++] << 24 +
                    //     data[offset++] << 16 +
                    //     data[offset++] << 8 +
                    //     data[offset++];

                    // tagFlags = data[offset++] << 8 +
                    //     data[offset++];

                    // tagStart = offset;

                    // owner should be 'com.apple.streaming.transportStreamTimestamp'
                    if (this._readUTF(data, offset, 44) === 'com.apple.streaming.transportStreamTimestamp') {
                        offset += 44;
                        // smelling even better ! we found the right descriptor
                        // skip null character (string end) + 3 first bytes
                        offset += 4;

                        // timestamp is 33 bit expressed as a big-endian eight-octet number, with the upper 31 bits set to zero.
                        let pts33Bit = data[offset++] & 0x1;
                        this._hasTimeStamp = true;

                        timestamp = ((data[offset++] << 23) +
                            (data[offset++] << 15) +
                            (data[offset++] << 7) +
                            data[offset++]) / 45;

                        if (pts33Bit) {
                            timestamp += 0x100000000;
                        }
                        timestamp = Math.round(timestamp);
                        this._dict['info'] = timestamp;
                        this._dict['type'] = 'transportStreamTimestamp';
                    } else {
                        this._dict['info'] = this._readUTF(data, offset, endPos - offset - 2);
                        this._dict['type'] = 'data';
                    }
                    break;
                default:
                    {
                        let size = this._getFrameSize(data, offset);
                        let result = null;
                        if (tagId[0] === 'T' && endPos > 20) {
                            result = this._decodeTextFrame(data, 20, tagId, size);
                        } else if (tagId[0] === 'W' && endPos > 20) {
                            result = this._decodeURLFrame(data, 20, tagId, size);
                        } else {
                            this._dict['info'] = 'Unknown';
                        }
                        if (result !== null && this._freeWheelIAB(tagId, result) === false) {
                            this._dict['info'] = result.info;
                            this._dict['data'] = result.data;
                        }
                    }
                    break;
            }
        }
    }

    _freeWheelIAB(tagId, tit) {
        let freeWheelAd = false;
        if (tagId === 'TIT2' || tagId === 'TIT3') {
            let freeWheelReg = new RegExp('.*(fw_(.*)_(firstQuartile|midPoint|thirdQuartile|complete|defaultImpression|default))');
            let result = freeWheelReg.exec(tit.data);
            if (result) {
                this._dict['id'] = result[2];
                this._dict['adType'] = result[3];
                this._dict['value'] = result[0]; // keep same with ios, for example, keep the '_' in _fw_9529098_318_defaultImpression.
                freeWheelAd = true;
            }
        }
        return freeWheelAd;
    }

    dict() {
        return this._dict;
    }
}
