import * as xhr from '../../Utils/Xhr'
import * as StringUtil from '../../Utils/String'
import * as ObjectUtil from '../../Utils/Object'
import { NLError } from '../NLError'

export class ServerCertificateManager {
    constructor() {
        this.MAX_COUNT = 2;
        // {key, value, async}
        //      |key|   is the cert url.
        //      |value| is ArrayBuffer.
        //      |async| is the Promise.
        this._cache = [];
    }

    load(url, token) {
        // Make sure 'session.async' count less than MAX_COUNT.
        if (this._pendingCount() >= this.MAX_COUNT) {
            return Promise.reject(NLError.Basic.CERTIFICATE_OVERFLOW_ERROR);
        }

        let certSession = this._hasCert(url);
        if (certSession) {
            return certSession.value ? Promise.resolve(certSession.value) : certSession.async;
        }

        let headers = {};
        headers['CustomData'] = token;
        headers['Authorization'] = token;

        let async = xhr.createXHR({
            url: url,
            headers: headers
        }).then(function (ab) {
            certSession.value = StringUtil.base64DecodeUint8Array(ab.replace(/(\r\n|\n|\r)/gm, ''));
            delete certSession.async;
            return certSession.value;
        }).catch(function (e) {
            this._removeSession(certSession);
            return Promise.reject(e);
        }.bind(this));

        certSession = { key: url, async: async };

        this._pushSession(certSession);
        return async;
    }

    destroy() {
        this._cache.forEach(function (item) {
            let async = item.async;
            async && async.abort && async.abort();
        });
    }

    _hasCert(url) {
        return ObjectUtil.find(this._cache, 'key', url);
    }

    _pendingCount() {
        let n = this._cache.length;
        let count = 0;
        for (let i = 0; i < n; i++) {
            if (this._cache[i].async) {
                count++;
            }
        }
        return count;
    }

    _pushSession(session) {
        if (this._cache.length >= this.MAX_COUNT) {
            this._cache.shift();
        }
        this._cache.push(session);
    }

    _removeSession(session) {
        let idx = this._cache.indexOf(session);
        if (idx >= 0) {
            this._cache.splice(idx, 1);
        }
    }
}
