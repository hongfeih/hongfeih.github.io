    import * as helpers from './FairPlayDRMHelpers'
import { EventManager } from '../../Manager/EventManager'
import * as ObjectUtil from '../../Utils/Object'
import * as StringUtil from '../../Utils/String'
import * as xhr from '../../Utils/Xhr'
import * as PlatformUtil from '../../Utils/Platform'
import { NLError } from '../NLError'

export class KeySessionManager {
    constructor() {
        this.KEY_SYSTEM_NAME = 'com.apple.fps.1_0';
        this._licenseUrl = '';
        this._onError = null;
        this._params = null;

        this._eventManager = new EventManager();
        this._sessions = [];
    }

    init(licenseUrl, params, onError) {
        this._licenseUrl = licenseUrl;
        this._params = params;
        this._onError = onError;

        this._eventManager.removeAll();
        this._clearSession();
        this._sessions = [];
    }

    createSession(videoElement, rawInitData, certCb) {
        if (!certCb) {
            // cert load failed, just return
            return;
        }

        let contentId = helpers.extractContentId(rawInitData);
        let overrideInitData = helpers.concatInitDataIdAndCertificate(
            rawInitData,
            contentId,
            new Uint8Array(certCb));

        videoElement.webkitSetMediaKeys(new WebKitMediaKeys(this.KEY_SYSTEM_NAME));

        let keySession = videoElement.webkitKeys.createSession('video/mp4', overrideInitData);
        if (!keySession) {
            this._onError({ error: NLError.Basic.DRM.KEY_SESSION_CREATE_ERROR });
            return;
        }
        this._eventManager.listen(keySession, 'webkitkeymessage', this._onKeyMessage.bind(this));
        this._eventManager.listen(keySession, 'webkitkeyerror', this._onKeyError.bind(this));

        this._sessions.push({ keySession: keySession, contentId: contentId });
    }

    reset() {
        this.init('', null, null);
    }

    destroy() {
        this.reset();
        this._eventManager.destroy();
        this._video = null;
    }

    _onKeyMessage(event) {
        let keySession = event.target;
        let session = this._getSession(keySession);

        let spcBase64 = StringUtil.base64EncodeUint8Array(event.message);
        let licenseServer = this._licenseUrl;

        let headers = {};
        let params = '';

        if (this._params) {
            // saffron request
            // {
            //  'content_id': '53504841B2694F515BCDCCE20810656E',
            //  'device_id':'8d025a8a-d0fa-47f4-8eaa-a4627d11e045',
            //  'challenge':'AAAAAQAAAACNCi9C3uGspgtxwmW�',
            //  'client':'Tribeca',
            //  'cro':'bearer eyJhbGciOiJIUzI1NiIsInR5cC�' 
            // }
            params = '{"content_id":"' + session.contentId + 
                '","device_id":"' + PlatformUtil.getDeviceId() + 
                '","challenge":"' + spcBase64 + 
                '","client":"' + (this._params.provider || 'NeuLion') + 
                '","cro":"' + this._params.token + '"}';

            headers['Content-type'] = 'application/x-www-form-urlencoded';
            headers['Content-Type'] = 'application/x-www-form-urlencoded';
            headers['Accept'] = 'application/json, text/plain, */*';
        }

        let async = xhr.createXHR({
            url: licenseServer,
            method: 'POST',
            headers: headers,
            params: params
        });

        async.then(function (responseText) {
            let key = helpers.parseLicense(responseText);
            if (!key.result) {
                this._onError({
                    error: {
                        code: NLError.Basic.DRM.LICENSE_ERROR.code,
                        message: NLError.Basic.DRM.LICENSE_ERROR.message + key.message
                    }
                });
            } else {
                delete session.async;
                keySession.update(key.result);
            }
        }.bind(this)).catch(function (e) {
            delete session.async;
            this._onError({ error: NLError.Basic.DRM.LICENSE_ERROR, code: e });
        }.bind(this));

        session.async = async;
    }

    _onKeyError(event) {
        this._onError({ error: NLError.Basic.DRM.KEY_ERROR });
    }

    _getSession(s) {
        return ObjectUtil.find(this._sessions, 'keySession', s);
    }

    _clearSession() {
        if (!this._sessions || this._sessions.length <= 0) {
            return;
        }

        this._sessions.forEach(function (s) {
            if (s.async) {
                s.async.abort();
            }
            if (s.keySession) {
                s.keySession.close();
            }
        });
    }
}
