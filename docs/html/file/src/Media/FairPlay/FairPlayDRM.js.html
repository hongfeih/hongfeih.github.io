    import { NLError } from '../NLError'
import { ServerCertificateManager } from './ServerCertificateManager'
import { KeySessionManager } from './KeySessionManager'
import { EventManager } from '../../Manager/EventManager'

export class FairPlayDRM {
    constructor() {
        this._clear();

        this._certManager = new ServerCertificateManager();
        this._eventManager = new EventManager();
        this._sessionManager = new KeySessionManager();
    }

    configure(licenseUrl, certUrl, params) {
        this._licenseUrl = licenseUrl;
        this._certUrl = certUrl;
        this._params = params;
    }

    init(video, onError) {
        this._video = video;
        this._onError = onError;

        this._sessionManager.init(
            this._licenseUrl,
            this._params,
            onError);

        if (typeof this._video.onwebkitneedkey !== 'undefined') {
            this._eventManager.listen(
                this._video,
                'webkitneedkey',
                this._onNeedKey.bind(this));
        } else {
            this._onError({ error: NLError.Basic.DRM.NOT_SUPPORTED });
        }
    }

    reset() {
        this._clear();

        // We don't want to reset CertManager because want to keep the cert in cache.
        this._eventManager && this._eventManager.removeAll();
        this._sessionManager && this._sessionManager.reset();
    }

    destroy() {
        this._clear();

        this._eventManager && this._eventManager.destroy();
        this._certManager && this._certManager.destroy();
        this._sessionManager && this._sessionManager.destroy();
    }

    _clear() {
        this._video = null;
        this._licenseUrl = '';
        this._certUrl = '';
        this._params = null;
        this._onError = null;
    }

    _onNeedKey(event) {
        if (!this._licenseUrl || !this._certUrl) {
            this.reset();
            this._onError({ error: NLError.Basic.DRM.INVALID_PARAMS });
            return;
        }

        this._certManager.load(this._certUrl, this._params.token).then(function (certCb) {
            this._sessionManager.createSession(this._video, event.initData, certCb);
        }.bind(this), function (e) {
            /* error :( */
            this._onError({ error: NLError.Basic.DRM.CERTIFICATE_ERROR, code: e });
        }.bind(this));
    }

    _onKeyError(event) {
        this._onError({ error: NLError.Basic.DRM.KEY_ERROR });
    }
}
