    src/Player/WebRTCPlayer.js

    import { AdaptivePlayer } from '../AdaptivePlayer';
import { PlayerBase } from './PlayerBase'
import { EventManager } from '../Manager/EventManager';
import { Settings } from '../config';
import { NLTrack } from '../Media/NLTrack';
import { FakeEvent } from '../Utils/FakeEvent';
import { NLError } from '../Media/NLError';

export default class WebRTCPlayer extends PlayerBase {
  constructor(video, parent) {
    super(video, parent);

    this._tracks = [];
    this._bitrate = 0;

    this._config = {};

    this.type = AdaptivePlayer.PlayerType.WEBRTC;
    this.streamType = AdaptivePlayer.StreamType.WEBRTC;

    this._wsCodeNormalClosed = Settings.WEBRTCPLAYER.WS_NORMAL_CLOSE;

    // register events from Shaka
    // manage events of video element
    this._eventManager = new EventManager();
  }

  configure(config) {
    this.print(AdaptivePlayer.LogLevel.DEBUG, 'WebRTCPlayer', 'configure called!');
    this._config = {
      iceServers: Settings.WEBRTCPLAYER.DEFAULT_ICE_SERVERS,
      abr: { enabled: true }
    };
    if (config.iceServers) {
      this._config.iceServers = config.iceServers;
    }
    this._config.abr = config.abr;
  }

  load(url, startTime) {
    this.print(AdaptivePlayer.LogLevel.DEBUG, 'WebRTCPlayer', 'load called!');
    this._url = url;
    if (this._websocket) {
      this._websocket.close(this._wsCodeNormalClosed);
    } else {
      this._websocket = new WebSocket(this._url);
      this._eventManager.listen(this._websocket, 'open', this._onWSOpened.bind(this));
      this._eventManager.listen(this._websocket, 'message', this._onWSMessage.bind(this));
      this._eventManager.listen(this._websocket, 'close', this._onWSClose.bind(this));
      this._eventManager.listen(this._websocket, 'error', this._onWSError.bind(this));
    }
  }

  isLive() {
    return (this._video.duration === Infinity);
  }

  unload() {
    this.print(AdaptivePlayer.LogLevel.DEBUG, 'WebRTCPlayer', 'unload called!');
    if (this._eventManager) {
      this._eventManager.removeAll();
    }
    if (this._peer_connection) {
      this._peer_connection.close();
    }
    if (this._video.captureStream) {
      this._video.captureStream().getTracks().forEach(function (track) { track.stop(); });
    }

    this._video.srcObject = null;
    this._url = '';
    this._current_bw = 0;
    this._bitrate = 0;
    this._tracks = [];

    return Promise.resolve();
  }

  destroy() {
    this.print(AdaptivePlayer.LogLevel.DEBUG, 'WebRTCPlayer', 'destroy called!');

    this.unload();

    if (this._eventManager) {
      this._eventManager.destroy();
    }
    return Promise.resolve();
  }

  getStats() {
    return {
      bytesLoaded: NaN,
      bandwidth: this._current_bw / 1000,
      bitrate: this._bitrate / 1000,
      switchInfo: NaN,
      switchHistory: NaN,
      isAutoSwitch: !!this._config.abr.enabled,
      livePointOffest: -1
    };
  }

  videoTracks() {
    return this._tracks;
  }

  selectTrack(track) {
    if (track) {
      let node = {};
      node.type = 'user_bitrate';
      node.payload = {};
      node.payload.bitrate = parseInt(track.bandwidth / 1000);
      let jsonString = JSON.stringify(node);

      this._websocket.send(jsonString);

      this.print(AdaptivePlayer.LogLevel.TRACE, 'WebRTCPlayer', 'auto quality switch status changed to manual');

      this._bitrate = parseInt(track.base);

      this._config.abr.enabled = false;
    }
  }

  setAdaptation(value) {
    this.print(AdaptivePlayer.LogLevel.DEBUG, 'WebRTCPlayer', 'auto quality switch status changed to auto');
    this._config.abr.enabled = value;
    if (value) {
      let node = {};
      node.type = 'user_bitrate';
      node.payload = {};
      node.payload.bitrate = parseInt(0);
      let jsonString = JSON.stringify(node);

      this._websocket.send(jsonString);
    }
  }

  notifyProfileChanged(data) {
    // notify profile change
    for (let i = 0; i < this._tracks.length; i++) {
      this._tracks[i].active = false;
      if (this._tracks[i].bandwidth === this._bitrate) {
        this._tracks[i].active = true;
        let newData = {
          id: i,
          videoWidth: data.videoWidth,
          videoHeight: data.videoHeight
        };
        this._eventHandler.dispatchEvent(new FakeEvent(AdaptivePlayer.EventType.ProfileChange, {
          details: newData
        }));
        this.print(AdaptivePlayer.LogLevel.DEBUG, 'WebRTCPlayer', 'Fired ProfileChange: ' + JSON.stringify(newData));
      }
    }
  }

  _onWSOpened(event) {
    this.print(AdaptivePlayer.LogLevel.DEBUG, 'WebRTCPlayer', 'ws opened');

    this._peer_connection = new RTCPeerConnection(this._config);
    this._peer_connection.ontrack = function (event) { // Never called in chrome 61
      this.print(AdaptivePlayer.LogLevel.DEBUG, 'WebRTCPlayer', 'track added');
    }.bind(this);

    this._peer_connection.onaddstream = function (event) {
      this.print(AdaptivePlayer.LogLevel.DEBUG, 'WebRTCPlayer', 'stream added');
      this._video.srcObject = event.stream;
    }.bind(this);
  }

  _onWSMessage(msg) {
    this.print(AdaptivePlayer.LogLevel.TRACE, 'WebRTCPlayer', 'message : ' + msg.data);

    let data = JSON.parse(msg.data);

    switch (data.type) {
      case 'offer': 
        this._peer_connection.setRemoteDescription({ type: 'offer', sdp: data.payload });

        this._peer_connection.onicecandidate = function (event) {
          if (event.candidate) {
            this._websocket.send(JSON.stringify({ type: 'ice', payload: event.candidate }));
          }
        }.bind(this);

        this._peer_connection.createAnswer().then(function (answer) {
          this._peer_connection.setLocalDescription(answer);

          this.print(AdaptivePlayer.LogLevel.TRACE, 'WebRTCPlayer', 'answer : ' + JSON.stringify(answer));

          this._websocket.send(JSON.stringify({ type: 'answer', payload: answer }));
        }.bind(this));
        break;

      case 'ice': 
        this.print(AdaptivePlayer.LogLevel.TRACE, 'WebRTCPlayer', 'ice candidate : ' + JSON.stringify(data.payload));
        this._peer_connection.addIceCandidate(new RTCIceCandidate(data.payload)).catch(function (e) {
          this.print(AdaptivePlayer.LogLevel.TRACE, 'WebRTCPlayer', 'add ice failed ' + e);
          this._onError(e);
        });
        break;

      case 'available_qualities': {
        this.print(AdaptivePlayer.LogLevel.TRACE, 'WebRTCPlayer', 'available qualities: ' + JSON.stringify(data.bitrates));

        let bitrates = data.bitrates;
        this._tracks = [];
        for (let i = 0; i < bitrates.length; i++) {
          this._tracks.push(new NLTrack('VIDEO', bitrates[i].bitrate * 1000, AdaptivePlayer.PlayerType.WEBRTC, i, this));
          this.print(AdaptivePlayer.LogLevel.TRACE, 'WebRTCPlayer', 'available qualities : ' + JSON.stringify(bitrates[i].bitrate + ' Kbit'));
        }

        this._websocket.send(JSON.stringify({ type: 'hello' }));

        this._bitrate = bitrates[0].bitrate * 1000;
      }
        break;

      case 'current_net_bw': 
        this.print(AdaptivePlayer.LogLevel.TRACE, 'WebRTCPlayer', 'Current Network Bandwidth: ' + data.bitrate + 'Kbit');
        this._current_bw = data.bitrate * 1000;
        break;

      case 'current_quality': 
        this.print(AdaptivePlayer.LogLevel.TRACE, 'WebRTCPlayer', 'current quality: ' + JSON.stringify(data.bitrate));
        this._bitrate = data.bitrate * 1000;
        break;
    }
  }

  _onWSClose(evt) {
    if (evt.code === WebRTCPlayer.WS_NORMAL_CLOSE) {
      this.print(AdaptivePlayer.LogLevel.DEBUG, 'WebRTCPlayer', 'ws closed');
    } else {
      this.print(AdaptivePlayer.LogLevel.DEBUG, 'WebRTCPlayer', 'ws connection error');
      this._onError(evt);
    }
    this._websocket = null;
  }

  _onWSError(evt) {
    if (this._websocket.readyState === 1) {
      this.print(AdaptivePlayer.LogLevel.DEBUG, 'WebRTCPlayer', 'ws normal error: ' + evt.type);
    }
  }

  _onError(e) {
    let level = NLError.SeverityLevel.Fatal;
    let errorCode = NLError.Code.Get_Primary_Manifest_Failed + NLError.NetworkDetailCode.Connect_Failed;
    let url = this._url;
    let message = 'WebRTC closeEvent ';

    // https://developer.mozilla.org/en-US/docs/Web/API/CloseEvent
    // TODO: Need more test to set correct errorCode
    if (e.code === 1000) { // Normal Closure
      // ;
    } else if (e.code === 1001) { //	Going Away 
      message += 'Going Away';
    } else if (e.code === 1002) { // Protocol Error
      message += 'Protocol Error';
    } else if (e.code === 1003) { //	Unsupported Data 
      message += 'Protocol Error';
    } else if (e.code === 1004) { // 
      message += 'Protocol Error';
    } else if (e.code === 1005) { //	No Status Recvd 
      message += 'No Status Recvd';
    } else if (e.code === 1006) { // Abnormal Closure
      message += 'Abnormal Closure';
    } else if (e.code === 1007) { //	Invalid frame payload data 
      message += 'Invalid frame payload data';
    } else if (e.code === 1008) { //	Policy Violation
      message += 'Policy Violation';
    } else if (e.code === 1009) { // Message too big
      message += 'Message too big';
    } else if (e.code === 1010) { //	Missing Extension
      message += 'Missing Extension';
    } else if (e.code === 1011) { // Internal Error
      message += 'Internal Error';
    } else if (e.code === 1012) { //	Service Restart 
      message += 'Service Restart ';
    } else if (e.code === 1013) { //	Try Again Later
      message += 'Try Again Later';
    } else if (e.code === 1014) { // Bad Gateway
      message += 'Bad Gateway';
    } else if (e.code === 1015) { //	TLS Handshake
      message += 'TLS Handshake';
    } else { //	Not defined yet 
      return;
    }

    let errorInfo = NLError.create(errorCode, level, url, message, e.code);
    this._statisticManager.reportError(errorInfo);
    this.print(AdaptivePlayer.LogLevel.ERROR, 'WebRTCPlayer', 'Fired streaming error: ' + message);
  }
}

WebRTCPlayer.version = 'v0.0.1';
